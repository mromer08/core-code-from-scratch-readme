# Exercise
Write a function that takes a string of parentheses, and determines if the order of the parentheses is valid. The function should return true if the string is valid, and false if it's invalid.
### Examples
```js
"()"              =>  true
")(()))"          =>  false
"("               =>  false
"(())((()())())"  =>  true
```
### Constraints

`0 <= input.length <= 100`


# Solution
This solution uses recursion
```js
function isValidParentheses(str, index, balance) {
  if (index === str.length) {
    return balance === 0;
  }

  if (str[index] === '(') {
    return isValidParentheses(str, index + 1, balance + 1);
  }

  if (str[index] === ')') {
    if (balance === 0) {
      return false;
    }
    return isValidParentheses(str, index + 1, balance - 1);
  }

  return isValidParentheses(str, index + 1, balance);
}

function validParentheses(parens) {
  return isValidParentheses(parens,0,0);
}
```
Another solution:

```js
function validParentheses(parens){
  while(parens.indexOf('()') != -1){
    parens = parens.replace('()', '');
  }
  return !parens.length;
}
```
```js
function validParentheses(parens){
  var n = 0;
  for (var i = 0; i < parens.length; i++) {
    if (parens[i] == '(') n++;
    if (parens[i] == ')') n--;
    if (n < 0) return false;
  }
  
  return n == 0;
}
```
